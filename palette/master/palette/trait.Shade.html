The Shade trait allows a color to be lightened or darkened.

use palette::{Rgb, Shade};

let a = Rgb::new(0.4, 0.4, 0.4);
let b = Rgb::new(0.6, 0.6, 0.6);

assert_eq!(a.lighten(0.1), b.darken(0.1));
